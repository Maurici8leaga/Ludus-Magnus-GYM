import React, { useState, useEffect } from 'react';
import PropTypes from 'prop-types';
import { connect } from 'react-redux';
import { Link } from 'react-router-dom';
import { getProfile, updateProfile } from '../../../../actions/profile';


const EditProfile = ({ getProfile, updateProfile, profile }) => {
    // los action y los props deben ir aca!!

    console.log('ESTE ES profile EN EDIT PROFILE', profile);


    const [formData, SetFormData] = useState({
        // creamos un state demanera de asi poder cambiar los valores de los inputs
        name: '',
        lastname: '',
        height: '',
        weight: ''
    });

    useEffect(() => {
        getProfile();
        
        SetFormData({
            // aca vamos a indicar que si la propiedad tiene un valor muestrala, si no dejala en ''
            name: profile.name ? profile.name : '',
            lastname: profile.lastname ? profile.lastname : '',
            height: profile.height ? profile.height : '',
            weight: profile.weight ? profile.weight : ''        
        });

        console.log('ESTE ES profile EN EDIT PROFILE', profile)

    }, [getProfile]);

    const { name, lastname, height, weight } = formData;
    // declaramos estas propiedas se almacenaran dentro de formData y este sera enviado al back

    const onChange = e => SetFormData({ ...formData, [e.target.name]: e.target.value });

    const onSubmit = e => {
        e.preventDefault();
        updateProfile(formData);
    }


    return (
        <div className="container-Profile-background">
            <div className="container-signin">
                <div>
                    <h2>Edita tu informacion</h2>
                </div>
                <form className="form" onSubmit={e => onSubmit(e)}>
                    <Link to="/profile" className="links-scss">
                        <i className="far fa-window-close"></i>
                    </Link>
                    <div>

                        <ul>
                            <li>
                                <p>Nombre</p>
                                <input className="form-input-individual" type="text" name="name" placeholder="Nombre" value={name} title="Nombre" onChange={e => onChange(e)}/>
                            </li>
                            <li>
                                <p>Apellido</p>
                                <input className="form-input-individual" type="text" name="lastname" placeholder="Apellido" value={lastname} onChange={e => onChange(e)}/>
                            </li>
                            <li>
                                <p>Peso</p>
                                <input className="form-input-individual" type="number" name="weight" placeholder="Kg" value={weight} onChange={e => onChange(e)}/>
                            </li>
                            <li>
                                <p>Altura</p>
                                <input className="form-input-individual" type="number" name="height" placeholder="Metros" value={height} onChange={e => onChange(e)}/>
                            </li>

                        </ul>

                        <button className="boton -primary-long" type="submit" value="Save"> Save </button>
                    </div>
                </form>
            </div>
        </div>


    )

}

EditProfile.propTypes = {
    getProfile: PropTypes.func.isRequired,
    updateProfile: PropTypes.func,
    profile: PropTypes.object.isRequired
    // ProfileUser: PropTypes.object
};


const mapStateToProps = state => ({
    // profile: state.profile.ProfileUser
});

export default connect(mapStateToProps, { getProfile, updateProfile })(EditProfile);
